//data.txt输入示例

mode = 0
num_list = 2
num_measure = [4,4]
num_b = [2,5]
x1 = [0.233599,0.283889,0.334319,0.384169];u_b = 0.001
x2 = [2.31101,2.79772,3.28168,3.76410];u_b = 0.00001
alpha = [1,-1]

//参数含义
mode:处理模式,取值为{single=0,multi=1,link=2}
    single：单变量处理模式，计算xi的各种观测值
    multi：当结果导出式g(xi)为形如g=Const*x_1^alpha_1*x_2^alpha_2*...*x_n^alpha_n乘积表达式时可使用，直接给出g的相对不确定度
    link：一元线性回归计算，输出相关系数，回归系数，以及数据分布导致的不确定度
num_list：所测量的数据组数量，即多少列xi
num_measure：各组测量数据数量，也就是每个xi有多少测量值，以列表形式给出，数值个数应等于num_list
num_b：单次测量值仪器测量次数，也就是每测量一次xi使用多少次仪器，造成了多少次测量误差，以列表形式给出，数值个数应等于num_list
xi：测量数据列表，以列表形式给出，数值个数应等于num_measure; u_bi：xi测量仪器的B类不确定度;两者在同一行被读取
alpha：模式multi积式指数，以列表形式给出，数值个数应等于num_list
